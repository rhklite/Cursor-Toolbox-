     def save_best_k_checkpoint(self, locs, iteration):
         if len(locs["rewbuffer"]) == 0:
             return
 
         if iteration < self.best_k_metric_start_iter:
             return
 
         selected_metric_value = statistics.mean(locs[self.best_k_metric_name])
         save_ckpt = False
         # keep tracking the best k checkpoints
         if len(self.best_k_metric_minheap) < self.best_k_metric_maxlen:
             heapq.heappush(self.best_k_metric_minheap, (selected_metric_value, iteration))
             save_ckpt = True
         else:
             if selected_metric_value > self.best_k_metric_minheap[0][0]:
                 heapq.heapreplace(self.best_k_metric_minheap, (selected_metric_value, iteration))
                 save_ckpt = True
 
         # save the best k checkpoints
         if save_ckpt:
             self.save(os.path.join(self.log_dir, f"model_{iteration}_best_k_value={selected_metric_value:.3f}.pt"))
             print(
                 f"Saved best k checkpoint at iteration {iteration} based on {self.best_k_metric_name}={selected_metric_value:.3f}"
             )

+

+    # - If torch.distributed is initialized: main process == rank 0
+    # - Otherwise: single process => main

+def sync_gradients(*modules: nn.Module) -> None:
+    """Synchronize gradients of multiple modules across all ranks.
+
+    Uses flatten-reduce-unfold pattern for efficiency (single collective op
+    per module instead of one per parameter).
+
+    No-op if not in distributed mode.
+
+    Note: If using Accelerate, prefer `accelerator.backward()` which handles
+    gradient sync automatically. This function is for manual gradient sync
+    in non-Accelerate setups.
+
+    Args:
+        *modules: One or more nn.Module instances whose gradients to synchronize.
+
+    Example:
+        >>> loss.backward()
+        >>> sync_gradients(actor_critic, discriminator)
+        >>> optimizer.step()
+    """
+    if not is_distributed():
+        return
+
+    world_size = dist.get_world_size()
+
+    for module in modules:
+        # Flatten all gradients into one tensor
+        grads = [p.grad.reshape(-1) for p in module.parameters() if p.grad is not None]
+        if not grads:
+            continue
+
+        all_grads = torch.cat(grads)
+
+        # All-reduce with average
+        dist.all_reduce(all_grads, op=dist.ReduceOp.SUM)
+        all_grads /= world_size
+
+        # Unfold back to each parameter
+        offset = 0
+        for param in module.parameters():
+            if param.grad is not None:
+                numel = param.numel()
+                param.grad.data.copy_(all_grads[offset : offset + numel].view_as(param.grad))
+                offset += numel

+def reduce_mean(tensor: torch.Tensor, accelerator: Optional["Accelerator"] = None) -> torch.Tensor:
+    """Compute the global mean of a tensor across all processes.
+
+    Args:
+        tensor: Input tensor to compute global mean for.
+        accelerator: Accelerator instance. If None, returns local mean.
+
+    Returns:
+        Scalar tensor containing the global mean.
+    """
+    if accelerator is None or accelerator.num_processes == 1:
+        return tensor.mean()
+
+    # Sum reduction across all processes
+    local_sum = tensor.sum()
+    local_count = torch.tensor(tensor.numel(), dtype=tensor.dtype, device=tensor.device)
+
+    # Stack for a single reduce call
+    stats = torch.stack([local_sum, local_count])
+    global_stats = accelerator.reduce(stats, reduction="sum")
+
+    global_sum, global_count = global_stats[0], global_stats[1]
+    return global_sum / global_count

+def standardize_tensor(
+    tensor: torch.Tensor,
+    accelerator: Optional["Accelerator"] = None,
+    eps: float = 1e-8,
+) -> torch.Tensor:
+    """Standardize a tensor to zero mean and unit variance.
+
+    If an Accelerator is provided and has multiple processes, computes global
+    statistics across all ranks using Accelerate's reduce operation.
+
+    This replaces the manual distributed advantage normalization logic in
+    RolloutStorage.compute_returns.
+
+    Args:
+        tensor: The tensor to standardize (e.g., advantages).
+        accelerator: Optional Accelerator instance for distributed training.
+            If None or single-process, performs local standardization.
+        eps: Small constant for numerical stability.
+
+    Returns:
+        Standardized tensor with approximately zero mean and unit variance.
+
+    Example:
+        >>> # In RolloutStorage.compute_returns:
+        >>> self.advantages = standardize_tensor(self.advantages, accelerator)
+    """
+    if accelerator is None or accelerator.num_processes == 1:
+        # Local normalization (single GPU)
+        return (tensor - tensor.mean()) / (tensor.std() + eps)
+
+    # Distributed: compute global mean and std using reduce
+    local_sum = tensor.sum()
+    local_count = torch.tensor(tensor.numel(), dtype=tensor.dtype, device=tensor.device)
+
+    # All-reduce to get global sum and count
+    stats = torch.stack([local_sum, local_count])
+    global_stats = accelerator.reduce(stats, reduction="sum")
+    global_sum, global_count = global_stats[0], global_stats[1]
+
+    global_mean = global_sum / global_count
+
+    # Compute global variance using Welford's parallel algorithm
+    local_sq_diff_sum = ((tensor - global_mean) ** 2).sum()
+    global_sq_diff_sum = accelerator.reduce(local_sq_diff_sum, reduction="sum")
+    global_var = global_sq_diff_sum / global_count
+    global_std = torch.sqrt(global_var + eps)
+
+    return (tensor - global_mean) / global_std

+class SyncRunningMeanStd(nn.Module):
+    """A transparent wrapper that adds distributed synchronization to a local RunningMeanStdTorch.
+
+    This follows the Minimum Invasion Principle: the underlying RunningMeanStdTorch
+    remains a simple, local-only implementation. Distributed behavior is added via
+    this wrapper layer.
+
+    The wrapper auto-detects distributed mode using is_distributed() and:
+    - In distributed mode: intercepts `update()` calls, computes local statistics,
+      synchronizes them globally using torch.distributed.all_reduce, then updates
+      the underlying instance with the global values.
+    - In single-process mode: acts as a transparent pass-through to the base instance.
+
+    Save/load works transparently via state_dict() since base is registered as a submodule.
+
+    Args:
+        base: The local RunningMeanStdTorch instance to wrap.
+
+    Example:
+        >>> from humanoid.algo.amp_ppo.utils import RunningMeanStdTorch
+        >>> base_normalizer = RunningMeanStdTorch(device, shape=(obs_dim,))
+        >>> sync_normalizer = SyncRunningMeanStd(base_normalizer)  # no accelerator needed!
+        >>> sync_normalizer.update(observations)  # Auto-syncs if distributed
+        >>>
+        >>> # Saving - just save directly
+        >>> torch.save({'normalizer': sync_normalizer.state_dict()}, 'checkpoint.pt')
+        >>> # Loading - just load directly
+        >>> sync_normalizer.load_state_dict(loaded_dict['normalizer'])
+    """
+
+    def __init__(self, base: nn.Module):
+        super().__init__()
+        self.base = base
+
+    @property
+    def mean(self) -> torch.Tensor:
+        """Access the running mean from the base instance."""
+        return self.base.mean
+
+    @property
+    def var(self) -> torch.Tensor:
+        """Access the running variance from the base instance."""
+        return self.base.var
+
+    @property
+    def count(self) -> torch.Tensor:
+        """Access the sample count from the base instance."""
+        return self.base.count
+
+    def forward(self, x: torch.Tensor) -> torch.Tensor:
+        """Forward pass delegates to base if it has a forward method."""
+        return self.base(x) if hasattr(self.base, 'forward') else x
+
+    @torch.jit.unused
+    def update(self, arr: torch.Tensor) -> None:
+        """Update running statistics with global synchronization if distributed.
+
+        In distributed mode: computes local batch statistics, synchronizes them
+        across all processes using torch.distributed.all_reduce, then updates
+        the base instance with global values.
+
+        In single-process mode: delegates directly to base.update().
+
+        Args:
+            arr: Tensor of shape (batch_size, *feature_dims) to update statistics with.
+        """
+        if not is_distributed():
+            # Pass-through to local update
+            self.base.update(arr)
+            return
+
+        # Compute local statistics with shifted algorithm for numerical stability
+        local_batch_size = arr.shape[0]
+        world_size = dist.get_world_size()
+        global_batch_size = world_size * local_batch_size
+
+        # Shift by current mean for numerical stability
+        x_shifted = arr - self.base.mean
+        local_sum_shifted = torch.sum(x_shifted, dim=0, keepdim=True)
+        local_sum_sq_shifted = torch.sum(x_shifted.pow(2), dim=0, keepdim=True)
+
+        # Sync stats across all processes (single all_reduce call for efficiency)
+        stats_to_sync = torch.cat([local_sum_shifted, local_sum_sq_shifted], dim=0)
+        dist.all_reduce(stats_to_sync, op=dist.ReduceOp.SUM)
+        global_sum_shifted, global_sum_sq_shifted = stats_to_sync[0], stats_to_sync[1]
+
+        # Compute global batch mean and variance
+        batch_mean_shifted = global_sum_shifted / global_batch_size
+        batch_var = global_sum_sq_shifted / global_batch_size - batch_mean_shifted.pow(2)
+        batch_mean = batch_mean_shifted + self.base.mean
+
+        # Update the base instance with global statistics
+        self.base.update_from_moments(batch_mean, batch_var, global_batch_size)
+

+        var = gsumsq / global_n - mean_shift.pow(2)

+@torch.no_grad()
+def broadcast_overwrite_from_src(
+    xs: Sequence[Union[np.ndarray, torch.Tensor]],
+    *,
+    src: int = 0,
+    device: Union[str, torch.device] = "cuda",
+    return_numpy: bool = False
+) -> Tuple[torch.Tensor, ...]:
+    """
+    Generic meta-broadcast that supports arbitrary shapes/dtypes.
+
+    Usage:
+        xs_out = broadcast_overwrite_from_src([a, b, c], src=0, device="cuda")
+        a, b, c = xs_out
+
+    Semantics:
+      - Must be called on ALL ranks when distributed.
+      - Only `src` rank's tensors/arrays are authoritative.
+      - Other ranks' inputs are ignored and may have different shapes/dtypes.
+      - Broadcasts metadata (dtype_code, ndim, shape) then payload for each item.
+      - Returns torch tensors on `device` (CUDA if device is cuda).
+
+    Preconditions:
+      - torch.distributed is initialized and world_size > 1 (guard with is_distributed()).
+      - `xs` length must be identical on all ranks (i.e., same number of items).
+      - Each item must be convertible to a torch tensor on src rank.
+    """
+    if not is_distributed():
+        raise RuntimeError("broadcast_overwrite_from_src() is distributed-only; guard it with is_distributed().")
+
+    dev = torch.device(device)
+    r = dist.get_rank()
+
+    # Ensure all ranks have same number of items.
+    # (We avoid Python-object broadcast for this; just assume caller passes same-length xs.)
+    k = len(xs)
+
+    out: List[torch.Tensor] = []
+
+    for i in range(k):
+        if r == src:
+            t_src = _to_device_tensor(xs[i], dev).contiguous()
+            code = _DTYPE_TO_CODE.get(t_src.dtype, None)
+            if code is None:
+                raise ValueError(f"Unsupported dtype for broadcast: {t_src.dtype}")
+            ndim = t_src.ndim
+        else:
+            t_src = None
+            code = 0
+            ndim = 0
+
+        # ---- broadcast (dtype_code, ndim) as CUDA int64 tensor ----
+        if r == src:
+            meta = torch.tensor([code, ndim], dtype=torch.int64, device=dev)
+        else:
+            meta = torch.empty(2, dtype=torch.int64, device=dev)
+        dist.broadcast(meta, src=src)
+
+        dtype = _CODE_TO_DTYPE[int(meta[0].item())]
+        ndim = int(meta[1].item())
+
+        # ---- broadcast shape (ndim int64) ----
+        if r == src:
+            shape_t = torch.tensor(list(t_src.shape), dtype=torch.int64, device=dev)  # type: ignore[union-attr]
+        else:
+            shape_t = torch.empty(ndim, dtype=torch.int64, device=dev)
+        dist.broadcast(shape_t, src=src)
+        shape = [int(v) for v in shape_t.tolist()]
+
+        # ---- allocate / coerce on src, allocate on others ----
+        if r != src:
+            t = torch.empty(shape, dtype=dtype, device=dev)
+        else:
+            t = t_src  # type: ignore[assignment]
+            if t.dtype != dtype:
+                t = t.to(dtype)
+            t = t.contiguous()
+
+        # ---- broadcast payload ----
+        dist.broadcast(t, src=src)
+        out.append(t.cpu().numpy() if return_numpy else t)
+
+    return tuple(out)

+def tensor_fingerprint_i64(t: torch.Tensor, stride: int = 1024) -> torch.Tensor:
+    """
+    Deterministic int64 fingerprint of tensor bits (subsampled).
+    Returns a 1-element int64 CUDA tensor.

+    # reinterpret raw bits -> signed integer stream

+        bits = x.view(torch.int32).to(torch.int64)
+    elif x.dtype in (torch.float16, torch.bfloat16):
+        bits = x.view(torch.int16).to(torch.int64)
+    elif x.dtype == torch.float64:
+        bits = x.view(torch.int64)
+    elif x.dtype in (torch.int64, torch.int32, torch.int16, torch.int8, torch.uint8, torch.bool):
+        bits = x.to(torch.int64)
+    else:
+        raise TypeError(f"unsupported dtype: {x.dtype}")
+
+    bits = bits[::stride]
+    const = 6364136223846793005  # odd constant; overflow is fine/deterministic
+    fp = (bits * const).sum(dtype=torch.int64)
+    return fp

     def slerp_sphere(self, val0, val1, blend):
         """Spherical linear interpolation between two values on a sphere surface."""
         dot = torch.sum(val0 * val1, dim=-1)
         dot = torch.clamp(dot, -1.0, 1.0)
         theta = torch.acos(dot) * blend
         sin_theta = torch.sin(theta)
         weight0 = torch.sin((1.0 - blend) * theta) / sin_theta
         weight1 = torch.sin(blend * theta) / sin_theta
         return weight0 * val0 + weight1 * val1

     def euler_to_quaternion(euler):
         """Convert euler angles(r,p,y) to quaternion(x,y,z,w)."""
         roll = euler[0]
         pitch = euler[1]
         yaw = euler[2]
 
         cy = math.cos(yaw * 0.5)
         sy = math.sin(yaw * 0.5)
         cp = math.cos(pitch * 0.5)
         sp = math.sin(pitch * 0.5)
         cr = math.cos(roll * 0.5)
         sr = math.sin(roll * 0.5)
 
         qw = cy * cp * cr + sy * sp * sr
         qx = cy * cp * sr - sy * sp * cr
         qy = sy * cp * sr + cy * sp * cr
         qz = sy * cp * cr - cy * sp * sr
 
         return np.array([qx, qy, qz, qw])
