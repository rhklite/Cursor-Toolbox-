+    def _update_mirror_index_offset(self):
+        if sys.version_info >= (3, 7):
+            obs_buf_dict, privileged_obs_buf_dict = self._get_obs_buf_dict()
+            self._set_mirror_offset(obs_buf_dict, self.cfg.mirror_cfg.obs_mirror)
+            self._set_mirror_offset(privileged_obs_buf_dict, self.cfg.mirror_cfg.privileged_obs_mirror)
+        else:
+            raise RuntimeError('python version < 3.7')

+    def _set_mirror_offset(self, obs_buf_dict, mirror_cfg):
+        obs_offset = 0
+        for key_name in obs_buf_dict:
+            value_len = obs_buf_dict[key_name].size(1)
+            mirror_attr = getattr(mirror_cfg, key_name)
+            setattr(
+                mirror_cfg,
+                key_name,
+                replace(mirror_attr, offset=obs_offset),
+            )
+            obs_offset += value_len

+    def _compute_torques(self, actions):
+        """Compute torques from actions.
+            Actions can be interpreted as position or velocity targets given to a PD controller, or directly as scaled torques.
+            [NOTE]: torques must have the same dimension as the number of DOFs, even if some DOFs are not actuated.
+
+        Args:
+            actions (torch.Tensor): Actions
+
+        Returns:
+            [torch.Tensor]: Torques sent to the simulation
+        """
+        actions_scaled = actions * self.cfg.control.action_scale
+        p_gains = self.p_gains * self.Kp_factors
+        d_gains = self.d_gains * self.Kd_factors
+        p_torques = p_gains * (actions_scaled + self.default_dof_pos - self.dof_pos)
+
+        if self.cfg.domain_rand.torque_delay:
+            self.torque_deque[:] = torch.cat([p_torques.unsqueeze(0), self.torque_deque[:-1]], dim=0)
+            p_torques = self.torque_deque[self.rand_torque_delay.long(), self.env_ids.long()]
+
+        d_torques = -d_gains * self.dof_vel
+        torques = p_torques + d_torques
+
+        torques = torch.clip(torques, -self.torque_limits, self.torque_limits)
+
+        return torques * self.motor_strengths
+
+    def _post_physics_step_callback(self):
+        super()._post_physics_step_callback()
+        self.traj_time_s += self.dt
+        self.gym.refresh_rigid_body_state_tensor(self.sim)
+        self.gym.refresh_jacobian_tensors(self.sim)
+        self.feet_pos[:], self.feet_vel[:] = self.foot_states_in_base_frame()
+        self.torso_quat[:] = self.rigid_state[:, self.torso_index, 3:7]
+        self.torso_ang_vel[:] = quat_rotate_inverse(self.torso_quat, self.rigid_state[:, self.torso_index, 10:13])
+        self.torso_projected_gravity[:] = quat_rotate_inverse(self.torso_quat, self.gravity_vec)
+
+        # update base velocity history
+        base_vel = torch.cat((self.base_lin_vel_yaw[:, 0:2], self.base_ang_vel[:, 2].unsqueeze(1)), dim=1)
+        # base_vel = torch.cat(
+        #     (self.root_states[:, 7:9], self.root_states[:, 12].unsqueeze(1)), dim=1
+        # )
+        self.base_vel_history = torch.cat((base_vel.unsqueeze(1), self.base_vel_history[:, :-1, :]), dim=1)
+        self.base_vel_history_sizes = torch.min(
+            self.base_vel_history_sizes + 1, torch.tensor(self.cfg.commands.average_stack, device=self.device)
+        )
+        self.base_avg_vel = torch.sum(self.base_vel_history, dim=1) / self.base_vel_history_sizes.unsqueeze(1)

+    def reset_idx(self, env_ids):
+        """Reset some environments.
+            Calls self._reset_dofs(env_ids), self._reset_root_states(env_ids), and self._resample_commands(env_ids)
+            [Optional] calls self._update_terrain_curriculum(env_ids), self.update_command_curriculum(env_ids) and
+            Logs episode info
+            Resets some buffers
+
+        Args:
+            env_ids (list[int]): List of environment ids which must be reset
+        """
+        t = str(int(time.time()))
+        for i in range(len(self.depth_cp)):
+            if self.cfg.env.record:
+                if not os.path.exists(self.cfg.env.record_path):
+                    os.makedirs(self.cfg.env.record_path)
+            np.save(self.depth_names[i] + t + "_" + str(self.common_step_counter) + "_depth.npy", self.depth_cp[i])
+            np.save(
+                self.depth_names[i] + t + "_" + str(self.common_step_counter) + "_height.npy",
+                self.last_height.cpu().numpy(),
+            )
+            np.save(
+                self.depth_names[i] + t + "_" + str(self.common_step_counter) + "_privileged_obs_history.npy",
+                self.privileged_obs_history[-1].cpu().numpy(),
+            )
+        self.depth_cp = []
+        self.depth_names = []
+        self.height_cp = []
+        self.reset_env_ids = env_ids
+        self.terminal_amp_states = self.get_amp_observations()[env_ids]
+
+        if len(env_ids) == 0:
+            return
+        # update curriculum
+        if self.cfg.terrain.curriculum:
+            self._update_terrain_curriculum(env_ids)
+        # avoid updating command curriculum at each step since the maximum command is common to all envs
+        if self.cfg.commands.curriculum and (self.common_step_counter % self.max_episode_length == 0):
+            self.update_command_curriculum(env_ids)
+
+        # reset robot states
+        if self.cfg.env.reference_state_initialization and self.cfg.env.amp_loader_type == 'BaseMotionLoader':
+            traj_idxs, traj_time_s, root_state, env_dof_pos, env_dof_vel = self.amp_loader.get_random_rsi_state_batch(
+                len(env_ids)
+            )
+            self._reset_dofs_motion(env_ids, env_dof_pos, env_dof_vel)
+            self._reset_root_states_motion(env_ids, root_state)
+            self.traj_idxs[env_ids] = traj_idxs
+            self.traj_time_s[env_ids] = traj_time_s
+        elif self.cfg.env.reference_state_initialization:
+            frames = self.amp_loader.get_full_frame_batch(len(env_ids))
+            self._reset_dofs_amp(env_ids, frames)
+            self._reset_root_states_amp(env_ids, frames)
+        else:
+            self._reset_dofs(env_ids)
+            self._reset_root_states(env_ids)
+
+        self.base_quat[env_ids] = self.root_states[env_ids, 3:7]
+        self.base_euler_xyz[env_ids] = get_euler_xyz_tensor(self.base_quat[env_ids])
+
+        self._resample_commands(env_ids)
+
+        self._randomize_dof_props(env_ids)
+
+        # reset buffers
+        self.gym.refresh_rigid_body_state_tensor(self.sim)
+        self.gym.refresh_jacobian_tensors(self.sim)
+        self.feet_pos[env_ids], self.feet_vel[env_ids] = self.foot_states_in_base_frame(env_ids)
+        self.actions[env_ids] = 0.0
+        self.last_actions[env_ids] = 0.0
+        self.last_rigid_state[env_ids] = 0.0
+        self.last_dof_vel[env_ids] = 0.0
+        self.feet_air_time[env_ids] = 0.0
+        self.episode_length_buf[env_ids] = 0
+        self.reset_buf[env_ids] = 1
+        # fill extras
+        self.extras["episode"] = {}
+        for key in self.episode_sums.keys():
+            self.extras["episode"]['rew_' + key] = (
+                torch.mean(self.episode_sums[key][env_ids]) / self.max_episode_length_s
+            )
+            self.episode_sums[key][env_ids] = 0.0
+        # log additional curriculum info
+        if self.cfg.terrain.mesh_type == "trimesh":
+            self.extras["episode"]["terrain_level"] = torch.mean(self.terrain_levels.float())
+        if self.cfg.commands.curriculum:
+            self.extras["episode"]["max_command_x"] = self.command_ranges["lin_vel_x"][1]
+        # send timeout info to the algorithm
+        if self.cfg.env.send_timeouts:
+            self.extras["time_outs"] = self.time_out_buf
+
+        # fix reset gravity bug
+        self.projected_gravity[env_ids] = quat_rotate_inverse(self.base_quat[env_ids], self.gravity_vec[env_ids])
+        self.torso_quat[env_ids] = self.rigid_state[env_ids, self.torso_index, 3:7]
+
+        self.torso_projected_gravity[env_ids] = quat_rotate_inverse(self.torso_quat[env_ids], self.gravity_vec[env_ids])
+
+        for i in range(self.obs_history.maxlen):
+            self.obs_history[i][env_ids] *= 0
+        for i in range(self.privileged_obs_history.maxlen):
+            self.privileged_obs_history[i][env_ids] *= 0
+
+        self.rand_torque_delay[env_ids] = torch.randint(
+            0, self.torque_deque_size, (len(env_ids),), dtype=torch.int32, device=self.device
+        )
+        self.torque_deque[:, env_ids].zero_()
+
+        self.rand_leg_linear_dof_pos_offset[env_ids, :][:, self.leg_linear_dof_indices] = (
+            torch_rand_float(
+                -1.0,
+                1.0,
+                (len(env_ids), len(self.leg_linear_dof_indices)),
+                device=self.device,
+            )
+            * self.leg_linear_dof_pos_offset_max
+        )
+        self.reset_base_vel_history(env_ids)
+
+    def reset_base_vel_history(self, env_ids):

+    def _reset_link_states_amp(self, env_ids, frames):
+        """Reset rigid link states (base, torso, feet) from AMP frames."""
+        # sync base link rigid state with root_state (index 0 is base link)
+        self.rigid_state[env_ids, self.baselink_index, 0:3] = self.root_states[env_ids, :3]
+        self.rigid_state[env_ids, self.baselink_index, 3:7] = self.root_states[env_ids, 3:7]
+        self.rigid_state[env_ids, self.baselink_index, 7:10] = self.root_states[env_ids, 7:10]
+        self.rigid_state[env_ids, self.baselink_index, 10:13] = self.root_states[env_ids, 10:13]
+
+        # optional rigid link positions are stored in dataset world frame; align them with current root offset
+        root_offset = self.root_states[env_ids, :3] - self.amp_loader.get_root_pos_batch(frames)
+
+        # torso rigid state (optional in dataset)
+        torso_pos, torso_quat, torso_lin, torso_ang = self.amp_loader.get_torso_rigid_state_batch(frames)
+        if torso_pos.shape[1] > 0:
+            self.rigid_state[env_ids, self.torso_index, 0:3] = torso_pos + root_offset
+        if torso_quat.shape[1] > 0:
+            self.rigid_state[env_ids, self.torso_index, 3:7] = torso_quat
+        if torso_lin.shape[1] > 0:
+            self.rigid_state[env_ids, self.torso_index, 7:10] = torso_lin
+        if torso_ang.shape[1] > 0:
+            self.rigid_state[env_ids, self.torso_index, 10:13] = torso_ang
+
+        # left foot
+        l_pos, l_quat, l_lin, l_ang = self.amp_loader.get_left_foot_rigid_state_batch(frames)
+        if l_pos.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[0], 0:3] = l_pos + root_offset
+        if l_quat.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[0], 3:7] = l_quat
+        if l_lin.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[0], 7:10] = l_lin
+        if l_ang.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[0], 10:13] = l_ang
+
+        # right foot
+        r_pos, r_quat, r_lin, r_ang = self.amp_loader.get_right_foot_rigid_state_batch(frames)
+        if r_pos.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[1], 0:3] = r_pos + root_offset
+        if r_quat.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[1], 3:7] = r_quat
+        if r_lin.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[1], 7:10] = r_lin
+        if r_ang.shape[1] > 0:
+            self.rigid_state[env_ids, self.feet_indices[1], 10:13] = r_ang
+
+        # self.gym.set_rigid_body_state_tensor(self.sim, gymtorch.unwrap_tensor(self.rigid_state))
+
+    def _process_rigid_body_props(self, props, env_id):
+        """
+        Add random mass and center of mass offsets to the rigid body properties.
+
+        Different mass and center of mass offsets are applied to the torso and the rest of the body.
+
+        Args:
+            props (list): List of rigid body properties.
+            env_id (int): ID of the environment.
+
+        Returns:
+            list: Updated list of rigid body properties.
+        """
+        link_mass_rand = torch_rand_float(
+            self.cfg.domain_rand.link_mass_range[0],
+            self.cfg.domain_rand.link_mass_range[1],
+            (len(props), 1),
+            device=self.device,
+        )
+        link_com_rand = torch_rand_float(
+            self.cfg.domain_rand.link_com_range[0],
+            self.cfg.domain_rand.link_com_range[1],
+            (len(props), 3),
+            device=self.device,
+        )
+        for i in range(len(props)):
+            if self.body_names[i] == self.cfg.asset.torso_name:
+                props[i].mass += self.payloads[env_id]
+                props[i].com += gymapi.Vec3(
+                    self.cfg.domain_rand.com_offset[0] + self.com_displacements[env_id, 0],
+                    self.cfg.domain_rand.com_offset[1] + self.com_displacements[env_id, 1],
+                    self.cfg.domain_rand.com_offset[2] + self.com_displacements[env_id, 2],
+                )
+            else:
+                props[i].mass *= link_mass_rand[i]
+                props[i].com = gymapi.Vec3(
+                    props[i].com.x * link_com_rand[i, 0],
+                    props[i].com.y * link_com_rand[i, 1],
+                    props[i].com.z * link_com_rand[i, 2],
+                )
+
+        return props
+
+    def get_amp_observations(self):
+        """
+        Construct the observations for the AMP environment
+
+        Returns:
+            torch.Tensor: The observations for the AMP environment
+        """
+        amp_tag_dim = len(self.cfg.env.amp_scene_tag_lists)
+        frame_scene_tags = torch.zeros(self.num_envs, amp_tag_dim, device=self.device)
+        if self.cfg.env.amp_add_scene_tags and self.cfg.env.amp_scene_tag_lists is not None:
+            stand_mask = torch.norm(self.commands[:, :3], dim=1) < 0.1
+            turn_mask = (
+                (torch.abs(self.commands[:, 0]) < 0.01) & (torch.abs(self.commands[:, 2]) > 0.01) & (~stand_mask)
+            )
+            straight_mask = (~stand_mask) & (~turn_mask)
+            for idx, tag_name in enumerate(self.cfg.env.amp_scene_tag_lists):
+                if tag_name == "stand":
+                    frame_scene_tags[stand_mask, idx] = 1
+                elif tag_name == "turn":
+                    frame_scene_tags[turn_mask, idx] = 1
+                elif tag_name == "straight":
+                    frame_scene_tags[straight_mask, idx] = 1
+
+        if self.cfg.env.amp_loader_type == 'BaseMotionLoader':
+            robot_motion_state = RobotMotionStateBase(
+                time_stamp_s=self.traj_time_s.unsqueeze(-1),
+                trajectory_id=self.traj_idxs.unsqueeze(-1),
+                base_pos_in_world_frame=self.root_states[:, :3] - self.env_origins,
+                base_quat_xyzw_in_world_frame=self.root_states[:, 3:7],
+                base_lin_vel_in_base_frame=self.base_lin_vel,
+                base_ang_vel_in_base_frame=self.base_ang_vel,
+                motion_joint_pos=self.dof_pos[:, self.amp_obs_joint_indices],
+                motion_joint_vel=self.dof_vel[:, self.amp_obs_joint_indices],
+                motion_link_pos_in_world_frame=self.rigid_state[:, self.amp_obs_link_indices, :3]
+                - self.env_origins.unsqueeze(1),
+                motion_link_quat_xyzw_in_world_frame=self.rigid_state[:, self.amp_obs_link_indices, 3:7],
+                motion_link_lin_vel_in_world_frame=self.link_vel_in_world_frame(self.amp_obs_link_indices),
+                motion_link_ang_vel_in_world_frame=self.link_ang_vel_in_world_frame(self.amp_obs_link_indices),
+                motion_link_contact_state=1.0
+                * (torch.norm(self.contact_forces[:, self.amp_obs_link_indices, :], dim=-1) > 0.1),
+            )
+            return self.amp_loader.construct_full_motion_obs(robot_motion_state)
+        else:
+            amp_state = R01AMPStateBase(
+                base_lin_vel=self.base_lin_vel,
+                base_ang_vel=self.base_ang_vel,
+                joint_pos=self.dof_pos,
+                joint_vel=self.dof_vel,
+                feet_pos=self.feet_pos,
+                feet_vel=self.feet_vel,
+                projected_gravity=self.projected_gravity,
+                frame_scene_tags=frame_scene_tags,
+            )
+            return self.amp_loader.construct_amp_observation(amp_state)
+
+    def _resample_commands(self, env_ids):
+        """
+        Randomly select commands of some environments.
+        (with certain chance of selecting the stop commands for each robot)
+        This function will override the same function name under legged_robot.py
+
+        Args:
+            env_ids (List[int]): Environments ids for which new commands are needed
+        """
+
+        super()._resample_commands(env_ids)
+
+        zero_linvel_prob = self.cfg.commands.zero_linvel_prob
+        zero_linvel_prob = min(1.0, max(0.0, zero_linvel_prob))
+        if (zero_linvel_prob > 0.0) and (env_ids.shape[0] > 0):
+            apply_zero_linvel = torch.rand(len(env_ids), device=self.device) < zero_linvel_prob
+            self.commands[env_ids, 0] = torch.where(
+                apply_zero_linvel, torch.zeros_like(self.commands[env_ids, 0]), self.commands[env_ids, 0]
+            )
+
+        stop_command_prob = self.cfg.commands.stop_command_prob
+        stop_command_prob = min(1.0, max(0.0, stop_command_prob))
+        # for each robot, randomly decide if the command should be zero based on the stop_command_prob
+        if (stop_command_prob > 0.0) and (env_ids.shape[0] > 0):
+            apply_stop_command = torch.rand(len(env_ids), device=self.device) < stop_command_prob
+            self.commands[env_ids[apply_stop_command], :] = 0.0
+
+        self.reset_base_vel_history(env_ids)
+
+    def foot_states_in_base_frame(self, env_ids=None):
+        if env_ids is None:
+            env_ids = slice(None)
+        rigid_body_state = self.rigid_state[env_ids]
+        last_rigid_state = self.last_rigid_state[env_ids]
+        root_quat_in_world_frame = self.root_states[env_ids, 3:7]
+        root_vel_in_world_frame = self.root_states[env_ids, 7:10]
+        root_ang_vel_in_world_frame = self.root_states[env_ids, 10:13]
+
+        feet_pos_l_world = rigid_body_state[:, self.feet_indices[0], 0:3] - rigid_body_state[:, 0, 0:3]
+        feet_pos_r_world = rigid_body_state[:, self.feet_indices[1], 0:3] - rigid_body_state[:, 0, 0:3]
+        feet_pos_l = quat_rotate_inverse(root_quat_in_world_frame, feet_pos_l_world)
+        feet_pos_r = quat_rotate_inverse(root_quat_in_world_frame, feet_pos_r_world)
+
+        feet_vel_l_in_world_frame = (
+            rigid_body_state[:, self.feet_indices[0], 0:3] - last_rigid_state[:, self.feet_indices[0], 0:3]
+        ) / self.dt
+        feet_vel_r_in_world_frame = (
+            rigid_body_state[:, self.feet_indices[1], 0:3] - last_rigid_state[:, self.feet_indices[1], 0:3]
+        ) / self.dt
+
+        feet_pos_l_cross = torch.cross(root_ang_vel_in_world_frame, feet_pos_l_world, dim=1)
+        feet_pos_r_cross = torch.cross(root_ang_vel_in_world_frame, feet_pos_r_world, dim=1)
+
+        feet_vel_l = quat_rotate_inverse(
+            root_quat_in_world_frame, feet_vel_l_in_world_frame - root_vel_in_world_frame - feet_pos_l_cross
+        )
+        feet_vel_r = quat_rotate_inverse(
+            root_quat_in_world_frame, feet_vel_r_in_world_frame - root_vel_in_world_frame - feet_pos_r_cross
+        )
+
+        return torch.cat((feet_pos_l, feet_pos_r), dim=1), torch.cat((feet_vel_l, feet_vel_r), dim=1)
+
+    def link_vel_in_world_frame(self, link_indices):
+        return (self.rigid_state[:, link_indices, 0:3] - self.last_rigid_state[:, link_indices, 0:3]) / self.dt
+
+    def link_ang_vel_in_world_frame(self, link_indices):
+        return quaternion_xyzw_diff(
+            self.rigid_state[:, link_indices, 3:7], self.last_rigid_state[:, link_indices, 3:7], self.dt
+        )

+    # ================================================ Rewards ================================================== #
+
+    def _reward_lin_vel_z(self):
+        # Penalize z axis base linear velocity
+        return torch.square(self.base_lin_vel[:, 2])
+
+    def _reward_ang_vel_xy(self):
+        # Penalize xy axes base angular velocity
+        return torch.sum(torch.square(self.base_ang_vel[:, :2]), dim=1)
+
+    def _reward_orientation(self):
+        # Penalize non flat base orientation
+        return torch.sum(torch.square(self.projected_gravity[:, :2]), dim=1)
+
+    def _reward_base_height(self):
+        # Penalize base height away from target
+        base_height = torch.mean(self.root_states[:, 2].unsqueeze(1) - self.measured_heights, dim=1)
+        return torch.square(base_height - self.cfg.rewards.base_height_target)
+
+    def _reward_torques(self):
+        # Penalize torques
+        return torch.sum(torch.square(self.torques), dim=1)
+
+    def _reward_torques_percent(self):
+        return torch.sum(torch.square(self.torques / self.torque_limits), dim=1)
+
+    def _reward_dof_vel(self):
+        # Penalize dof velocities
+        return torch.sum(torch.square(self.dof_vel), dim=1)
+
+    def _reward_feet_vel(self):
+        # Penalize feet velocities
+        return torch.sum(torch.square(self.feet_vel), dim=1)
+
+    def _reward_dof_acc(self):
+        # Penalize dof accelerations
+        return torch.sum(torch.square((self.last_dof_vel - self.dof_vel) / self.dt), dim=1)
+
+    def _reward_action_rate(self):
+        # Penalize changes in actions
+        return torch.sum(torch.square(self.last_actions - self.actions), dim=1)
+
+    def _reward_foot_distance_limit(self):
+        # Penalize foot distance limit violation
+        return (
+            torch.norm(self.feet_pos[:, :3] - self.feet_pos[:, 3:], dim=1) - self.cfg.rewards.foot_distance_threshold
+        ).clip(max=0.0) * -1.0
+
+    def _reward_cross_feet(self):
+        # Penalize cross feet
+        right_foot_y = self.feet_pos[:, 1]
+        left_foot_y = self.feet_pos[:, 4]
+        return (left_foot_y - right_foot_y - 0.05).clip(max=0.0) * -1.0
+
+    def _reward_collision(self):
+        # Penalize collisions on selected bodies
+        return torch.sum(
+            1.0 * (torch.norm(self.contact_forces[:, self.penalised_contact_indices, :], dim=-1) > 0.1), dim=1
+        )
+
+    def _reward_termination(self):
+        # Terminal reward / penalty
+        return self.reset_buf * ~self.time_out_buf
+
+    def _reward_dof_pos_limits(self):
+        # Penalize dof positions too close to the limit
+        out_of_limits = -(self.dof_pos - self.soft_dof_pos_limits[:, 0]).clip(max=0.0)  # lower limit
+        out_of_limits += (self.dof_pos - self.soft_dof_pos_limits[:, 1]).clip(min=0.0)  # upper limit
+        return torch.sum(out_of_limits, dim=1)
+
+    def _reward_dof_vel_limits(self):
+        # Penalize dof velocities too close to the limit
+        # clip to max error = 1 rad/s per joint to avoid huge penalties
+        return torch.sum(
+            (torch.abs(self.dof_vel) - self.soft_dof_vel_limits).clip(min=0.0),
+            dim=1,
+        )
+
+    def _reward_torque_limits(self):
+        # penalize torques too close to the limit
+        return torch.sum((torch.abs(self.torques) - self.soft_torque_limits).clip(min=0.0), dim=1)
+
+    def _reward_tracking_avg_lin_vel(self):
+        # Tracking of linear velocity commands (xy axes) against the average linear velocity of the base
+        lin_vel_error = torch.sum(torch.square(self.commands[:, :2] - self.base_avg_vel[:, :2]), dim=1)
+        return torch.exp(-lin_vel_error / self.cfg.rewards.tracking_sigma_lin_vel)
+
+    def _reward_tracking_avg_ang_vel(self):
+        # Tracking of angular velocity commands (yaw) against the average angular velocity of the base
+        ang_vel_error = torch.square(self.commands[:, 2] - self.base_avg_vel[:, 2])
+        return torch.exp(-ang_vel_error / self.cfg.rewards.tracking_sigma_ang_vel)
+
+    def _reward_tracking_lin_vel(self):
+        # Tracking of linear velocity commands (xy axes)
+        lin_vel_error = torch.sum(torch.square(self.commands[:, :2] - self.base_lin_vel[:, :2]), dim=1)
+        return torch.exp(-lin_vel_error / self.cfg.rewards.tracking_sigma_lin_vel)
+
+    def _reward_tracking_ang_vel(self):
+        # Tracking of angular velocity commands (yaw)
+        ang_vel_error = torch.square(self.commands[:, 2] - self.base_ang_vel[:, 2])
+        return torch.exp(-ang_vel_error / self.cfg.rewards.tracking_sigma_ang_vel)
+
+    def _reward_feet_air_time(self):
+        # Reward long steps
+        # Need to filter the contacts because the contact reporting of PhysX is unreliable on meshes
+        contact = self.contact_forces[:, self.feet_indices, 2] > 1.0
+        contact_filt = torch.logical_or(contact, self.last_contacts)
+        self.last_contacts = contact
+        first_contact = (self.feet_air_time > 0.0) * contact_filt
+        self.feet_air_time += self.dt
+        rew_airTime = torch.sum(
+            (self.feet_air_time - 0.2) * first_contact, dim=1
+        )  # reward only on first contact with the ground
+        rew_airTime *= (torch.norm(self.commands[:, :3], dim=1) > 0.1) * 2.0 - 1.0  # no reward for zero command
+        self.feet_air_time *= ~contact_filt
+        return rew_airTime
+
+    def _reward_stumble(self):
+        # Penalize feet hitting vertical surfaces
+        return torch.any(
+            torch.norm(self.contact_forces[:, self.feet_indices, :2], dim=2)
+            > 5 * torch.abs(self.contact_forces[:, self.feet_indices, 2]),
+            dim=1,
+        )
+
+    def _reward_stand_still(self):
+        # Penalize motion at zero commands
+        return torch.sum(torch.abs(self.dof_pos - self.standing_dof_pos), dim=1) * (
+            torch.norm(self.commands[:, :3], dim=1) < 0.1
+        )
+
+    def _reward_feet_contact_forces(self):
+        # penalize high contact forces
+        return torch.sum(
+            (
+                torch.norm(self.contact_forces[:, self.feet_indices, :], dim=-1) - self.cfg.rewards.max_contact_force
+            ).clip(min=0.0),
+            dim=1,
+        )
+
+    def _reward_default_joint_pos(self):
+        """
+        Calculates the reward for keeping joint positions close to default positions, with a focus
+        on penalizing deviation in yaw and roll directions. Excludes yaw and roll from the main penalty.
+        """
+        joint_diff = self.dof_pos - self.default_dof_pos
+        left_yaw_roll = joint_diff.index_select(1, torch.tensor([0, 2], device=self.device))
+        right_yaw_roll = joint_diff.index_select(1, torch.tensor([6, 8], device=self.device))
+        yaw_roll = torch.norm(left_yaw_roll, dim=1) + torch.norm(right_yaw_roll, dim=1)
+        yaw_roll = torch.clamp(yaw_roll - 0.1, 0, 50)
+        return torch.exp(-yaw_roll * 100)  # - 0.01 * torch.norm(joint_diff, dim=1)
+
+    def _reward_energy(self):
+        return torch.sum(torch.abs(torch.multiply(self.torques, self.dof_vel)), dim=1)
+
+    def _reward_torque_rate(self):
+        """
+        Penalizes the rate of change of the torques in the robot's joints. This encourages smooth and controlled
+        movements by limiting the acceleration of the robot's joints.
+        """
+        return torch.sum(torch.square((self.last_torques - self.torques) / self.torque_limits / self.dt), dim=1)
+
+    # ================================================ Mirror ================================================== #
+
+    def _init_mirror_index_lists(self):
+        """
+        Initialize the mirror index lists for the observations, privileged observations, and actions.
+        """
+        (
+            self.mirror_obs_inverse_index,
+            self.mirror_obs_exchange_src_list,
+            self.mirror_obs_exchange_dst_list,
+        ) = self._get_combined_index_lists(self.cfg.mirror_cfg.obs_mirror)
+        (
+            self.mirror_privileged_obs_inverse_index,
+            self.mirror_privileged_obs_exchange_src_list,
+            self.mirror_privileged_obs_exchange_dst_list,
+        ) = self._get_combined_index_lists(self.cfg.mirror_cfg.privileged_obs_mirror)
+        (
+            self.mirror_action_inverse_index,
+            self.mirror_action_exchange_src_list,
+            self.mirror_action_exchange_dst_list,
+        ) = self._get_combined_index_lists(self.cfg.mirror_cfg.action_mirror)
+
+    def _get_combined_index_lists(self, mirror_index_class):
+        """get the combined index lists(inverse, exchange left, exchange right) for mirror_obs or mirror_privileged_obs
+
+        Args:
+            mirror_index_class (R01AMPCfg.mirror_cfg.obs_mirror or R01AMPCfg.mirror_cfg.privileged_obs_mirror): config class for mirror index generation
+
+        Returns:
+            inverse_list (list): list of inverse indices
+            exchange_left_list (list): list of exchange source indices
+            exchange_right_list (list): list of exchange destination indices
+        """
+        inverse_list = []
+        exchange_left_list = []
+        exchange_right_list = []
+        for attr_name in dir(mirror_index_class):
+            attr = getattr(mirror_index_class, attr_name)
+            if hasattr(attr, 'offset') and all(
+                hasattr(attr, key) for key in ['inverse_index', 'exchange_src_index', 'exchange_dst_index']
+            ):
+                offset = attr.offset
+                inverse_list.extend([i + offset for i in attr.inverse_index])
+                exchange_left_list.extend([i + offset for i in attr.exchange_src_index])
+                exchange_right_list.extend([i + offset for i in attr.exchange_dst_index])
+        return (
+            torch.tensor(inverse_list, device=self.device),
+            torch.tensor(exchange_left_list, device=self.device),
+            torch.tensor(exchange_right_list, device=self.device),
+        )
+
+    def get_symmetric_states_actions(
+        self, obs: torch.Tensor, actions: torch.Tensor, obs_type: str = "policy"
+    ) -> torch.Tensor:
+        """get the symmetric states and actions
+
+        Args:
+            obs (torch.tensor): observations or privileged observations
+            actions (torch.tensor): actions
+            obs_type (str, optional): type of observations. "policy" for observations, "critic" for privileged observations. Defaults to "policy".
+
+        Returns:
+            obs (torch.tensor): symmetric observations or privileged observations
+            actions (torch.tensor): symmetric actions
+        """
+        if obs is not None:
+            if obs_type == "policy":
+                mirror_obs = torch.empty_like(obs)
+                mirror_obs[...] = obs
+                mirror_obs = mirror_obs.view(-1, self.cfg.env.frame_stack, self.cfg.env.num_single_obs).contiguous()
+                mirror_obs[:, :, self.mirror_obs_exchange_dst_list] = mirror_obs[
+                    :, :, self.mirror_obs_exchange_src_list
+                ]
+                mirror_obs[:, :, self.mirror_obs_inverse_index] *= -1
+                mirror_obs = mirror_obs.view(-1, self.cfg.env.num_observations)
+                obs = torch.cat([obs, mirror_obs], dim=0)
+            elif obs_type == "critic":
+                mirror_obs = torch.empty_like(obs)
+                mirror_obs[...] = obs
+                mirror_obs[:, self.mirror_privileged_obs_exchange_dst_list] = mirror_obs[
+                    :, self.mirror_privileged_obs_exchange_src_list
+                ]
+                mirror_obs[:, self.mirror_privileged_obs_inverse_index] *= -1
+                obs = torch.cat([obs, mirror_obs], dim=0)
+
+        if actions is not None:
+            mirror_action = torch.empty_like(actions)
+            mirror_action[..., self.mirror_action_exchange_dst_list] = actions[
+                ..., self.mirror_action_exchange_src_list
+            ]
+            mirror_action[..., self.mirror_action_inverse_index] *= -1
+            actions = torch.cat([actions, mirror_action], dim=0)
+
+        return obs, actions
