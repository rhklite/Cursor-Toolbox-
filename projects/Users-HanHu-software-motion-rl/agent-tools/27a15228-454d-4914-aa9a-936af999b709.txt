+    def assemble_obs_buffer(
+        self,
+        obs_proprio_history,
+        obs_extero_history,
+        privileged_obs_history,
+        privileged_obs_extero_history,
+        random_coef,
+    ):
+        """Assemble the observations and privileged observations into a single tensor."""
+        # ---- obs_buf (actor) ----
+        delay = random.randint(0, self.cfg.env.itof_delay_std) + self.cfg.env.itof_delay
+        length_obs_buf = obs_proprio_history.maxlen - self.cfg.env.itof_delay - self.cfg.env.itof_delay_std
+        T = length_obs_buf
+        K = self.cfg.env.num_frames_extero
+        B = obs_proprio_history[0].shape[0]
+        Dp = obs_proprio_history[0].shape[1]
+        De = obs_extero_history[0].shape[1]
+        total_obs_dim = T * Dp + K * De
+        obs_buf = obs_proprio_history[0].new_empty(B, total_obs_dim)
+
+        col = 0
+        for i in range(T):
+            p = obs_proprio_history[i + delay]  # [B, Dp]
+            obs_buf[:, col : col + Dp] = p
+            col += Dp
+            if T - i <= K:
+                e = obs_extero_history[i]  # [B, De]
+                if not torch.any(e):
+                    n = len(obs_extero_history)
+                    for j in range(i + 1, n):
+                        cand = obs_extero_history[j]
+                        if torch.any(cand):
+                            e = cand
+                            break
+                obs_buf[:, col : col + De] = e
+                col += De
+
+        # ---- privileged_obs_buf (critic) ----
+        c = self.cfg.env.c_frame_stack
+        B = privileged_obs_history[0].shape[0]
+        D_hist = privileged_obs_history[0].shape[1]
+        D_ext = privileged_obs_extero_history[0].shape[1]
+
+        total_priv_dim = c * (D_hist + D_ext)
+        buf = privileged_obs_history[0].new_empty(B, total_priv_dim)
+
+        col = 0
+        for i in range(c):
+            p = privileged_obs_history[i]  # [B, D_hist]
+            e = privileged_obs_extero_history[-c + i]  # [B, D_ext]

+            col += D_hist

+            col += D_ext

+        return obs_buf, privileged_obs_buf
